\subsection*{Key Findings:}

\textbf{Best Overall Result:}
\begin{itemize}
\item \textbf{Gemini 3 Flash APBF 3h = 144.89 kW} (54\% better than 315.82 kW baseline!)
\end{itemize}

\textbf{Best Model (by average):}
\begin{itemize}
\item \textbf{Gemini 3 Flash} (378.49 kW avg) - most consistent performance
\item Gemini 3 Pro (588.47 kW avg)
\item Claude 4.5 Haiku (602.54 kW avg) - limited data
\end{itemize}

\textbf{Best Strategy (by average):}
\begin{itemize}
\item \textbf{Naive} (535.62 kW avg) - surprisingly effective baseline
\item Advanced (573.47 kW avg) - mixed results
\item APBF (623.86 kW avg) - best peaks but inconsistent (excellent at 3h, poor at other horizons)
\end{itemize}

\textbf{Easiest Horizon:}
\begin{itemize}
\item \textbf{6h} (384.66 kW avg for Naive) - best for simple prompting
\item 3h (651.32 kW avg for Naive, but 144.89 with APBF!)
\item 48h (570.89 kW avg) - harder but not worst
\end{itemize}

\textbf{Models Beating Baseline (315.82 kW):}
\begin{itemize}
\item Gemini 3 Flash APBF 3h: 144.89 kW (best!)
\item Gemini 3 Flash APBF 6h: 211.87 kW
\item Claude 4.5 Haiku Naive 6h: 222.80 kW
\item Gemini 3 Flash Advanced 6h: 280.28 kW
\item Claude 4.6 Sonnet Naive 3h: 284.56 kW
\end{itemize}

\textbf{Strategy Effectiveness:}
\begin{itemize}
\item Advanced beats Naive for Gemini 3 Flash (421.92 vs 535.16 avg)
\item APBF excels at short horizons (3h, 6h) but struggles at 48h
\item Physics-informed prompting helps most for medium horizons (6h)
\end{itemize}
